import datetime as dt

end_datetime = dt.datetime.today()

al_res = await DomoActivityLog.get_activity_log(
    auth = token_auth,
    start_time = start_datetime,
    end_time = end_datetime,
    object_type = ActivityLog_ObjectType.ACTIVITY_LOG,
    maximum =5)
